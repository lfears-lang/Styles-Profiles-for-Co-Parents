function calculateTotals() {

let totals = [0,0,0,0];

const rows = document.querySelectorAll("#scoreTable tbody tr");

rows.forEach(row => {

row.querySelectorAll("input").forEach((input, idx) => {

const val = parseInt(input.value);

if (!isNaN(val)) totals[idx] += val;

});

});

// Update totals

["A","B","C","D"].forEach((letter, idx) => {

document.getElementById("total" + letter).textContent = totals[idx];

});

// Map to styles

document.getElementById("styleA").textContent = totals[0];

document.getElementById("styleB").textContent = totals[1];

document.getElementById("styleC").textContent = totals[2];

document.getElementById("styleD").textContent = totals[3];

// Highlight highest

const maxScore = Math.max(...totals);

["styleA","styleB","styleC","styleD"].forEach((id, idx) => {

const cell = document.getElementById(id);

if (totals[idx] === maxScore && maxScore > 0) {

cell.parentElement.classList.add("highlight");

} else {

cell.parentElement.classList.remove("highlight");

}

});

// Show interpretations

if (maxScore > 0) {

const topStyles = ["A","B","C","D"].filter((_, idx) => totals[idx] === maxScore);

const interpDiv = document.getElementById("interpretations");

interpDiv.innerHTML = topStyles.map(letter => `

<div class="style-title">${styleInfo[letter].name}</div>

${styleInfo[letter].desc}

`).join("");

document.getElementById("result").style.display = "block";

} else {

document.getElementById("result").style.display = "none";

} else {

document.getElementById("result").style.display = "none";

}

}

</script>
